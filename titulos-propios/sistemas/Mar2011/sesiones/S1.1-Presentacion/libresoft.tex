\begin{frame}
\frametitle{The Universidad Rey Juan Carlos}

\begin{itemize}
\item Young public university
\item Four \emph{campi} in Madrid: Vicálvaro, Alcorcón, Móstoles and Fuenlabrada
\item Around 2,000 staff (1,200 academics); 27,000 students
\item Structured in faculties: ours is the \emph{Escuela Superior de Ingenieros
de Telecomunicación}
\item Official degrees: now moving to the Bologna system
\begin{itemize}
  \item Degree (240 ECTS)
  \item Master (60-120 ECTS)
  \item PhD
\end{itemize}
\item Own titles
\end{itemize}

\begin{flushright}
\url{http://www.urjc.es/}
\end{flushright}

\end{frame}

\begin{frame}
\frametitle{Madrid On Rails (MoR)}

\begin{itemize}
\item Main goal: disseminate and promote the use of emerging technologies within small and medium companies in Madrid.
\item Specific goals:
\begin{itemize}
  \item Become a center of international reference in technology, emerging applications and services based on Free Software.
  \item Undertake innovation projects of national and international collaboration with universities, research institutes and companies
  \item Provide specialized training to SMEs in the emerging disciplines of free software and web 2.0, both onsite and online.
\end{itemize}
\end{itemize}

\begin{flushright}
\url{http://madridonrails.com/}
\end{flushright}

\end{frame}



%%---------------------------------------------------------------

\begin{frame}
\frametitle{The GSyC/LibreSoft group}

\begin{itemize}
\item Research group in Uni. Rey Juan Carlos (Fuenlabrada, Spain)
\item About 25 researchers and support staff (including students)
\item Focus on libre (free, open source) technologies, from several
  points of view
\item Backgrounds on software development, data communications,
  telecommunications, etc.
\item Involvement in 6th and 7th FP projects (CALIBRE, FLOSSInclude,
  FLOSSMETRICS, QUALOSS, QUALIPSO, Alert), national projects (Morfeo,
  Vulcano), projects
  funded by companies (Vodafone, Telefonica, Infojobs, Google, Andago) and competitive
  tenders (European Commission, CENATIC)
\end{itemize}

\begin{flushright}
\url{http://libresoft.es/}
\end{flushright}

\end{frame}



%%---------------------------------------------------------------

\begin{frame}
\frametitle{Main activity lines (GSyC/LibreSoft)}

\begin{itemize}
\item Study of libre software development (quantitative, empirical
  approach)
\item Study of Wikipedia and other on-line communities
\item Study and assessment of libre software communities (specially
  those promoted by companies, or composed in part by companies)
\item Technologies for software development (specially those designed
  for practices found in libre software projects)
\item Integration of mobile devices with the Internet
\item Augmented reality in several fields
\item Web 2.0 technologies and the social Internet
\item {\bf Server architecture with free software: Virtualization, clustering and cloud computing}
\end{itemize}

\end{frame}
