\begin{frame}
\frametitle{Key design idea(s)}

\begin{center}
{\Large
``To obtain in one year the competences that a current
  expert in libre software acquired by experience during the last 10-15
  years''
}
\end{center}

\begin{itemize}
\item Holistic approach
\item In line with other experiences in Europe and worldwide
\item One step ahead of ``just technical skills'' (but not forgetting
  technical issues)
\item Rooted in real world experiences and academic evidence
\item Let's explore new learning processes
\end{itemize}

\end{frame}

%%---------------------------------------------------------------

\begin{frame}
\frametitle{Some design assumptions}

\begin{itemize}
\item Knowledge about libre software is maturing, facts are available
\item Industry and society need professionals well trained in libre software
\item Target profiles: IT engineers (informatics, telecom, etc.),
  but open to other backgrounds
\item Goal: comprehensive, multidisciplinary education on the
  matter \\
  ``To reach, as a result of the master, the level of an
  expert with good knowledge of the libre software world''
\item All materials available as libre documents
\item Global community: English for written communication
\item Student activities as open as possible
\end{itemize}
\end{frame}

\begin{frame}

\begin{itemize}
\item Meeting point around \emph{libre software} in Madrid
\begin{itemize}
  \item Academia

  \item (Politics)
\end{itemize}
\item The \emph{libre software} way, e.g., synchronous (IRC) and asynchronous (mailing lists, forums) communication channels

%--------------------------------------------------------
\begin{frame}
\frametitle{Blended learning}

\begin{itemize}
\item Combination of good practices of on-site and off-site learning
\item Suites well to the geo-chracteristics of the region of Madrid...
\item ...and the lack of time availability of students
\item Theoretical lessons ~25\%: on-site ~80\%
\item Practical lessons ~60\%): on-site ~30\%
\item Evaluation ~15\%: on-site ~30\%
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Structure of the program}

General structure (September - May):

\begin{itemize}
\item 30 ECTS of mandatory teaching subjects
\item 6 ECTS of elective teaching subjects
\item 24 ECTS of Practicum (12) and Final Project (12)

\begin{frame}
\frametitle{Subject: Legal Aspects}

\begin{itemize}
\item Lecturers:
   \begin{itemize}
   \item Miguel Vidal (coord.)
   \item Gregorio Robles
   \end{itemize}
\item Why?: Licenses are legal basis and ``social contract'' of
libre software.
\item Main topics:
   \begin{itemize}
    \item Intellectual property (IP) law for non-lawyers: Copyright,
Patents, Trademarks, Warranties...
    \item How FLOSS and IP work together
    \item Free and Open Source Software licenses (permissive and reciprocal)
    \item Free cultural works (documentation, images, Creative Commons
licenses)
    \item Case studies and practical issues (choosing a license, 
compatibility,
forking, merging/mixing... myths)
   \end{itemize}
\end{itemize}

\end{frame}
