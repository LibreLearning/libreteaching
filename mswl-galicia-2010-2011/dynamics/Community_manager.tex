\section{Community Manager}

\begin{frame}
 \frametitle{What is a community manager?}
\begin{itemize}
 \item \textit{The online community organizer (or community manager) is one of the most important jobs of the future}
  \begin{flushright}
   Seth Godin          
  \end{flushright}
 \item Who is Seth Godin? \\
  \textit{The ultimate entrepreneur for the information age}
   \begin{flushright}
     Business Week               
   \end{flushright}
 \item Not sure what it means, but I'd like to be one of them!
\end{itemize}
\end{frame}


\begin{frame}
 \frametitle{What is a community manager?}
\begin{itemize}
  \item It is a figure useful for both, FLOSS communities, but also corporations
  \item Highly correlated to the FLOSS development model (distributed by definition)
  \item Hybrid business models where everyone could potentially participate
  \item Someone has to link all this together
\end{itemize}
\end{frame}


\begin{frame}
 \frametitle{What is a community manager?}
\begin{itemize}
  \item Cares for the health of the community
  \item Interacts with users
  \item Interacts with developers
  \item Interacts with stakeholders
  \item Organizational aspects of the project
  \item Is the liaison between the company and the community
  \item Different hats in different moments
\end{itemize}
\end{frame}


\section{Rationale}

\begin{frame}
 \frametitle{Rationale}
 \begin{itemize}
  \item Companies want to establish communities around their products
  \item They want people to engage with the company
  \item \textit{Community managers work with a number of stakeholders and perform many different activities. 
  They have no power over contributors and have to find interesting incentives, such as contributing 
  to a meaningful project, in order to motivate contributors.}
 \begin{flushright}
   Martin Mychlmier                 
  \end{flushright}
 \end{itemize}

\end{frame}


\begin{frame}
 \frametitle{Roles of a Community Manager}
 \begin{itemize}
  \item Facilitation.- Connect people, get started.
  \item Delegation.- Motivation
  \item Marketing.- Advocacy, press, conferences
  \item Listening and observation.- Information flow
  \item Creating vision.- Shared culture
  \item Reporting.- Community <=> Company
 \end{itemize}

\end{frame}


\begin{frame}
 \frametitle{Skills of a Community Manager}
 \begin{itemize}
  \item Communication
  \item Marketing
  \item  Management
  \item Technical Skills
  \item Patience and helpfulness
  \item Being visible and present
 \end{itemize}

\end{frame}



\begin{frame}
 \frametitle{Building community}
 \begin{itemize}
  \item Face to face meetings
  \item Stands at conferences
  \item Online training events
  \item Bug squashing parties
  \item Newsletter, planet of blogs
 \end{itemize}

\end{frame}
