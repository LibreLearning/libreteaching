\section{Public Administrations and Libre Software}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%---------------------------------------------------------------

\begin{frame}
\frametitle{Impacto en las administraciones públicas}

\begin{itemize}
\item Aprovechamiento más adecuado de recursos: más impacto para la
  misma inversión (ejemplo: localización)
\item Fomento del tejido tecnológico local
\item Independencia del proveedor
\item Adaptación a las necesidades exactas
\item Escrutinio público (en seguridad, rendimiento, etc.)
\item Disponibilidad a largo plazo
\end{itemize}

\end{frame}

%%---------------------------------------------------------------

\begin{frame}
\frametitle{Dificultades de adopción}

\begin{itemize}
\item Desconocimiento y falta de decisión política
\item Poca adecuación de los mecanismos de contratación
\item Falta de estrategia de implantación
\item Escasez o ausencia de productos libres en ciertos segmentos
\end{itemize}

\end{frame}

%%---------------------------------------------------------------

\begin{frame}
\frametitle{Actuaciones de las administraciones públicas}

\begin{itemize}
\item Compra de servicios y sistemas basados en software libre
  (genéricos, a medida, etc.)
\item Promoción de la sociedad de la información (uso de software a
  gran escala)
\item Fomento de la investigación
\end{itemize}

\end{frame}

%%---------------------------------------------------------------

\begin{frame}
\frametitle{Algunos escenarios ejemplo}

\begin{itemize}
\item Concurso público para la adquisición de un juego de aplicaciones
  ofimáticas con ciertas especificaciones
  \begin{itemize}
  \item proyecto a dos años,
  \item decenas o centenas de miles de puestos
  \item varios ganadores que
  compitan entre ellos por tener más usuarios
\end{itemize}
\item Consorcio de administraciones que contratan un software a medida
  con la condición de que el resultado sea software libre
  \begin{itemize}
  \item despliegue y otros servicios incluidos en el contrato
  \item otras administraciones interesadas pueden entrar en el
    consorcio o contratar por su cuenta
  \end{itemize}
\end{itemize}

\end{frame}
