\begin{frame}
\frametitle{Files}
\begin{itemize}
\item Source code must have several ``burocratic'' files: \\
License (LICENSE or COPYING)\\
General Info (README)\\
Changes made (Changelog) \\
Installation process info (INSTALL) \\
Documentation (Users, developers)\\
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Packages or binaries}
\begin{itemize}
\item They should be marked with the correct version
\item They must follow the same standard
\end{itemize}
\end{frame}
