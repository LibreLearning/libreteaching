\begin{frame}
	\frametitle{El software}

	\begin{itemize}
		\item Conjunto de instrucciones que le indican a la CPU lo
		que debe hacer.
		\item Objetivo: resolver problemas.
    		\item Separación: Datos + Instrucciones
		\item Totalmente flexible.
	\end{itemize}
\end{frame}

\section{¿Cómo se hace el SW?}

\begin{frame}	
	\frametitle{Algoritmos}

	\begin{itemize}
		\item Conjunto de pasos o procedimientos a realizar para llevar a cabo una
		tarea.
		\item Generalmente expresado en lenguaje natural.
		\item Ambiguo.
	\end{itemize}
\end{frame}

%%

\begin{frame}	
	\frametitle{Ejemplo de Algoritmo: El chef automático}

	\begin{enumerate}
		\item Mezclar 2 huevos {\it levemente} batidos  con una {\it cucharadita}  de
			vainilla,media de canela y una {\it taza} de leche.
		\item {\it Mojar} 6 rebanadas de pan en la mezcla.
		\item Freir un {\it  poco} de mantequilla {\it hasta que se doren}.
		\item Servir el pan con {\it  un poco de sirope} de caramelo.
	\end{enumerate}
	Todo lo que está en cursiva es ambiguo.
\end{frame}

%%

\begin{frame}	 
	\frametitle{Programas}

	\begin{itemize}
		\item Descripción formal de un algoritmo expresado en un lenguaje 
			de programación.
		\item Se comienza con un algoritmo y se acaba con un programa.		
		\item Los programas tienen que ser precisos.
		\item Ambigüedad: fuente de errores en los programas informáticos.
		\item Es común que tengan errores (bug). 
		\item Se prueban y validan (depuración, debug).
	\end{itemize}
\end{frame}


%%

\begin{frame}
	\frametitle{Lenguajes de programación}

	\begin{itemize}
		\item Lenguajes para implementar algoritmos y crear programas.
		\item Según nivel de abstracción:
			\begin{itemize}
				\item Lenguajes de bajo nivel: código máquina, ensamblador.				
				\item Lenguajes de alto nivel: Java, C, C++, ADA, PHP, Python...
			\end{itemize}
		\item Cuanto más nivel de abstracción, más se alejan de la máquina y 
			más se acercan al lenguaje natural.
	\end{itemize}
\end{frame}

%%

\begin{frame}	
	\frametitle{Código máquina y ensamblador}

	\begin{itemize}
		\item Se usa directamente el conjunto de instrucciones de la CPU.
		\item Ejemplo, código máquina: 11000100001010010100010
		\item Ejemplo, ensamblador: 	add \$s0 \$1 \$2
		\item ¡Muy cercano a la máquina!
	\end{itemize}
\end{frame}

%%

\begin{frame}	
	\frametitle{Lenguajes de alto nivel (I)}

	\begin{itemize}
		\item Permiten al programador centrarse en la implementación del
			algoritmo y olvidarse de las particularidades de la máquina.
		\item Ejemplo, C : a = b + c;
		\item Menos complejos, menos propensos a errores.
	\end{itemize}
\end{frame}

\section{El sistema operativo}

\begin{frame}	
	\frametitle{El sistema operativo (SO)}

		Si mi programa funciona con un ratón USB, ¿tengo que volver a escribirlo para poder
		usarlo con un \emph{touchpad}? 
		
	\begin{itemize}	
		\item Abstracción de la máquina: ofrece los dispositivos de la máquina a las aplicaciones
			ofreciendo abstracciones (ficheros, procesos, dispositivos, etc.).
		\item Reparto de recursos: hace que múltiples aplicaciones puedan usar la máquina {\it al
			mismo tiempo}.
	\end{itemize}
\end{frame}

%%

\begin{frame}		
	\frametitle{Aplicaciones para un sistema operativo}

	\begin{itemize}	
		\item Las aplicaciones están construidas para interactuar con un SO en concreto.		
		\item Un programa construido para un SO no puede ejecutar en un SO no-compatible.
		\item Puede haber versiones de un programa para SO incompatibles, pero no son realmente
			el mismo programa (aunque su parte fundamental sí lo sea).
	\end{itemize}
\end{frame}

\begin{frame}		
	\frametitle{Tareas del sistema operativo}

	\begin{itemize}	
		\item Comunicación con periféricos/dispositivos (\emph{drivers}).
		\item Gestión de memoria principal (memoria virtual).

		\item Monitorización y seguridad (permisos, registros, etc.).
		\item Gestión de almacenamiento (sistemas de ficheros).
		\item Coordinación de comunicaciones (gestión de la red).
	\end{itemize}
\end{frame}

\section{Software dañino}
		
\begin{frame}		
	\frametitle{Software {\it peligroso}}
	\begin{itemize}	
		\item Virus: programa {\it pegado} a otro software 
		que se instala sin nuestro consentimiento, se 
		modifica para evitar su detección y se propaga clonándose.
		\item Los hay inofensivos, pero también los hay muy dañinos.
		\item Transportes favoritos: correos, aplicaciones gratuitas, floppys.
	\end{itemize}
\end{frame}
			

%%

\begin{frame}		
	\frametitle{Software {\it peligroso}}
	\begin{itemize}	
		\item Troyano: programa dañino disfrazado de software común. Además de hacer
		lo que se supone que hace, se dedica a {\it otras cosas}.
		\item Gusano: programa dañino que se  replica y propaga por la red de forma
		automática aprovechando fallos en el software.
		\item Antivirus: programa que tiene una base de datos con los virus conocidos
			y sabe como desinstalarlos de nuestro sistema.	\end{itemize}
\end{frame}

\section{Licencias}


\begin{frame}		
	\frametitle{Licencias}
	Copyright: Conjunto de normas y principios que regulan los derechos
			morales y patrimoniales que la ley concede a los autores por el hecho
			de la creación de una obra literaria, artística o científica, tanto publicada
			como por publicar. El titular goza de los siguientes derechos exclusivos:
	\begin{itemize}	
		\item Reproducción de copias.
		\item Preparación de obras derivadas.
		\item Distribución y venta.
		\item Presentación pública.
	\end{itemize}
\end{frame}

%%

\begin{frame}		
	\frametitle{Licencias de software}
	Se rigen por la ley del copyright.
	Contratos que determina la cesión por parte del autor al usuario del derecho de uso
	de sus bienes. Determina:
	\begin{itemize}	
		\item Copia.
		\item Instalación.
		\item Transferencia a terceros.
		\item Modificación.
		\item Acceso al fuente.
	\end{itemize}
\end{frame}

%%


\begin{frame}		
	\frametitle{Licencias de software libre}
	Garantiza cuatro libertades básicas usando las leyes del copyright.
	\begin{itemize}	
		\item Libertad 0: usarlo para cualquier fin.
		\item Libertad 1: modificar y estudiar el programa.
		\item Libertad 2: copiarlo y distribuirlo.
		\item Libertad 3: distribuir trabajos derivados.
	\end{itemize}

	¡Ojo! libre NO significa gratuito. 

	Una licencia de software libre puede ser \emph{copyleft} cuando obliga a que todas las versiones futuras del software tengan que tener una licencia de software libre.

\end{frame}

\begin{frame}		
	\frametitle{Licencias libres (información)}
	Licencias libres para documentos e información:
	\begin{itemize}	
		\item Las licencias de software libre no son aplicables a textos, música, etc.
		\item Son licencias equivalentes para información en general, un complemento a
		las licencias de software libre.
		\item Creative Commons: {\it Algunos derechos reservados}.
		\item Son una serie de licencias marco. 
	\end{itemize}
\end{frame}

\begin{frame}		
	\frametitle{Licencias libres (información)}
	CC: Cuatro derechos que el autor puede retener o no:
	\begin{itemize}	
		\item Reconocimiento: se permite copia, distribución, muestra y representación
		siempre que le dé crédito.
		\item No comercial: '' ''  siempre que tengan fines
		no comerciales. 
		\item No trabajos derivados:   '' ''  pero no de los trabajos  derivados.
		\item Compartir bajo las mismas condiciones: se permite la redistribución de
		trabajos derivados siempre que se mantenga la licencia con la que lo reciben.
	\end{itemize}
\end{frame}
