\section{¿Qué hacen los ordenadores?}

\begin{frame}
	\frametitle{¿Qué hacen los ordenadores?}
	\begin{itemize}
  		\item Un ordenador trabaja únicamente con \alert{información}. Cuatro funciones básicas: 
    			\begin{itemize}
				\item Recibe \alert{información}.
				\item Procesa la \alert{información} recibida (entrada).
				\item Almacena la \alert{información}.
				\item Produce \alert{información} (salida).
			\end{itemize}
	\end{itemize}
\end{frame}

\section{Sobre la información}

\begin{frame}
	\frametitle{En este tema intentaremos responder a:}
	\begin{itemize}
		\item ¿Cómo definimos \alert{información}?
		\item ¿Qué tipo de información se puede \alert{almacenar y procesar}?
    	\item ¿La información tiene \alert{"tamaño"}?.
		\item ¿Qué significa que "ocupa" más o menos?
		\item ¿Es lo mismo hablar de \alert{datos} que de \alert{programas}?
      \item ¿Una misma información se puede representar de varias formas?
      \item ¿Es mejor una representación que otra?
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Sobre la información}
	\begin{itemize}
		\item El ordenador \alert{no entiende} de noticias, música o arte. El ordenador se alimenta de 
			datos para posteriormente procesar y almacenar la información obtenida.
		\item Cuando un ordenador almacena una determinada información lo que hace 
			es transformar ésta a una representación que pueda procesar.
	\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Instrucciones}
\begin{itemize}
	\item La CPU tiene un conjunto de operaciones: 
	\begin{itemize}
      		\item Aritméticas (suma, multiplica, etc.)
		\item Lógicas (compara este valor con otro, etc.) 
		\item Movimiento de información dentro del ordenador, etc.
	\end{itemize}
	\item Los programas que usamos son secuencias de este tipo de instrucciones.
	\item Dependen del fabricante de la CPU (Intel, AMD, Motorola, IBM...).
\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Medidas de información (tradicionales)}
	\begin{itemize}
		\item Bit: La unidad de información.
		\item Byte: Un conjunto de 8 bits.
		\item Kilobyte: Se representa por KB y es el equivalente de 1024 bytes. $2^{10}$.
		\item Megabyte: Se representa por MB y es el equivalente de 1024 KB. $2^{20}$.
		\item Gigabyte: Se representa por GB y es el equivalente de 1024 MB. $2^{30}$.
		\item Terabyte: Se representa por TB y es el equivalente de 1024 GB. $2^{40}$.
	\end{itemize}
\end{frame}

%%


\begin{frame}
	\frametitle{Medidas de información (tradicionales)}
	\begin{itemize}
		\item Cuando queremos pasar de una unidad inferior a otra 
			superior debemos dividir por 1024.
		\item Cuando queremos pasar de una unidad superior a 
			otra inferior debemos multiplicar por 1024.
	\end{itemize}
\end{frame}

\section{Ficheros}

\begin{frame}
\frametitle{Ficheros (archivos)}
	 \begin{itemize}
		\item Secuencia de bytes en un dispositivo de almacenamiento.
		\item Independientes entre sí, por lo que podemos tratarlos de 
			manera individual (copia, mover, borrado, etc.).
		\item Contiene los datos en un formato dado, dependiendo del
			programa que lo generó
		\item Hay formatos comunes, usados por muchos programas (p.e. UNICODE).
		\item Otros son propietarios y cerrados (p.e. Mapas de los GPS).
	\end{itemize}
\end{frame}
 
 
 
 %%
 
 
\begin{frame}
\frametitle{Ficheros (archivos)}
\begin{itemize}
	\item Se identifican mediante un nombre, que puede tener extensión.
 	\item A veces el tipo de fichero se refleja en la extensión, pero no siempre.
	\item Otras veces, se añade información del tipo al propio fichero (cabeceras, número mágico...).
	\item Sistema de ficheros (del Sistema Operativo).
	\item Atributos: fecha y hora de creación, modificación, permisos.
 	\end{itemize}
\end{frame}
 

%%


\begin{frame}
\frametitle{Compresión}
\begin{itemize}
  	\item Compresor: programa empleado para reducir el tamaño de un fichero.
    	\item Existen dos tipos de compresión:
    	\begin{itemize}
    		\item Sin pérdida: rar, zip...
        		\item Con pérdida: jpeg, mp3...
	\end{itemize}
	\item ¿Cuándo podemos usar cada uno? 
\end{itemize}	
\end{frame}


%%


\begin{frame}
 	\frametitle{Analógico y digital}
	\begin{itemize}
		\item Información en forma de señal eléctrica.
		\item Señal Analógica: Señal que puede tomar cualquier valor (continuo).
		\item Señal Digital: Señal  que sólo  puede tomar una serie de valores definidos (discreto).
		\item Digitalización: pasar de analógico a digital, ejemplo: audio.
		\end{itemize}
\end{frame}

%%
		
\begin{frame}
 	\frametitle{Analógico y digital}
	\begin{itemize}		
		\item Hoy en día los computadores son digitales: representan 
			los datos con valores discretos (bit).
		\item Computadoras analógicas: representan la información 
			(p.e. números) mediante mediciones de cantidades 
			físicas continuas (p.e. voltaje, ángulos, longitudes). 
			No son comunes desde 1960.
	\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Regla General}
\begin{itemize}
	\item Con el tiempo:
	\begin{itemize}
  		\item Se reduce el tamaño.
    		\item Se incrementa la potencia de cálculo y almacenamiento.
		\item Se incrementa su fiabilidad.
		\item Se reduce el precio.
	\end{itemize}	
	\item Economía de escala.
	\item Ley de Moore: el número de transistores en un procesador se
	 duplica cada 18-24 meses.
\end{itemize}
\end{frame}
