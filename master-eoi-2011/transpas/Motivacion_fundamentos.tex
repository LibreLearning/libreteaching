\begin{frame}
	\frametitle{Aplicaciones útiles}
	\begin{itemize}
  		\item Algunas aplicaciones de apoyo a la docencia (p.e. Moodle)
		\item FaceBook, Twitter,..
		\item Aplicaciones Google
		\item Spotify,..
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Representación de la información}
	\begin{itemize}
  		\item Algunos conceptos básicos: bit, byte, Kb, Mb, etc.
		\item Compresión de la información
      \item Formatos
      \item Conceptos de imagen, sonido, etc. (por ejemplo resolución, pixel, etc.)
		\item Preparación para el tema de cifrado y seguridad 
	\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Hardware}
	\begin{itemize}
  		\item ¿Cómo funcionan los ordenadores? (no queremos convertirnos en ingenieros)
		\item ¿Funcionan igual un ordenador portátil y el móvil? ¿y una consola de juegos? ¿y un controlador de temperatura del frigorífico?
		\item Algunos dispositivos de entrada/salida básicos
      \item ¿Qué cambia al tener un ordenador en red? 
      \item ¿Qué podemos quitarle a un ordenador para que siga siendo un ordenador? ¿Podemos quitarle el disco? ¿y el teclado?
      \item Conceptos básicos: CPU, memoria, almacenamiento, etc
	\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Software}
	\begin{itemize}
  		\item ¿Qué es un programa?
		\item ¿Por qué no es compatible mi programa?
      \item ¿Es importante el Sistema Operativo?
	   \item ¿Cómo se hace un programa?
		\item ¿Es diferente un algoritmo de un programa? ¿Es relevante?
      \item Conceptos básicos: código fuente, compilación, librerias, etc.
	\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Redes}
	\begin{itemize}
  		\item ¿Qué es una red? 
		\item ¿Todas las redes son tipo Internet? ¿Qué significa esto?
      \item ¿Qué alternativas hay a Internet?
		\item Conceptos básicos: ancho de banda, WAN, LAN, etc.
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Internet y Web}
	\begin{itemize}
      \item ¿Qué hace diferente a Internet?
	   \item ¿Cómo funciona Internet?
      \item ¿Internet es lo mismo que Web?
      \item Conceptos básicos: TCP/IP, paquetes, Http, Intranet/Internet, etc.
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Ejercicio}
	\begin{itemize}
	   \item Buscad en la wikipedia los siguientes conceptos (o temas):
\begin{itemize}
      \item  Bit y Byte
      \item  Compresión
      \item  Imagen
      \item  Hardware
      \item  Sistema operativo
      \item  Desarrollo software
      \item  Redes de ordenadores
\end{itemize}
      \item Cread ocho grupos de trabajo. Usad herramientas colaborativas para organizaros (Intentad hacerlo sin hablar).
		\item Cada grupo debe elegir uno de estos puntos sin repetirse.
      \item Preparad un documento en Google Docs que motive ese concepto.
      \item Tenemos 15 minutos para preparanos. Al terminar se lo contaremos a nuestros compañeros en público (Tendremos el tiempo limitado).
      \item Habrá una pequeña recompensa al que mejor nos motive :-)
	\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{Ejercicio}
	Cosas que podríamos contar para motivar a nuestros compañeros:
	\begin{itemize}
	   \item  ¿Por qué puede interesarnos este concepto o tema en este máster?
		\item  Mostrar un caso, ejemplo, producto, polémica o juicio relacionado con el concepto.
 		\item Existen aspectos externos a la tecnología que hace interesante el concepto (por ejemplo aspectos legales, sociales, etc.)
		\item Relación con otros conceptos del temario (tener un conocimiento de este concepto nos permite entender ..)
		\item ¿Crees que la tecnología asociada a dicho concepto puede cambiar con el tiempo? ¿Por qué? ¿Qué podemos esperar en el futuro?.
		\item Ha habido o podría haber futuras oportunidades de negocio relacionadas con ese tema.
	\end{itemize}
\end{frame}
