\begin{frame}[fragile]
\frametitle{Un servidor web muuuuuuy sencillo}

\begin{itemize}
\item \url{servidor-http-simple.py}

\begin{verbatim}
import socket
mySocket = socket.socket(socket.AF_INET,
                         socket.SOCK_STREAM)
mySocket.bind(('localhost', 1234))
mySocket.listen(5)
while True:
    (recvSocket, address) = mySocket.accept()
    print recvSocket.recv(1024)
    recvSocket.send("HTTP/1.1 200 OK\r\n\r\n" +
      "<html><body><h1>Hello!</h1></body></html>" +
      "\r\n")
    recvSocket.close()
\end{verbatim}

\item Para ejecutarlo:

\begin{verbatim}
python servidor-http-simple.py
\end{verbatim}

\end{itemize}

\end{frame}

\begin{frame}[fragile]
\frametitle{¿Cómo hablamos con él?}

\begin{itemize}
\item Pues desde un navegador... (probamos con Firefox)
\item O desde un plugin específico (probamos Web Developer de Firefox)
\item Pero también podemos ver qué nos dice

\item O hablar HTTP ``a pelo'' (desde un telnet)
\begin{verbatim}
telnet localhost 1234
\end{verbatim}

\item O pedirle al amigo curl que hable por nosotros
\begin{verbatim}
curl http://localhost:1234/
\end{verbatim}

\item O al amigo wget
\begin{verbatim}
wget http://localhost:1234/
\end{verbatim}

\item Y podemos espiar cualquier conversación con wireshark...
\end{itemize}

\end{frame}
